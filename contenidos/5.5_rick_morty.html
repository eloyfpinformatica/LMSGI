5.5. Trabajo con Datos JSON ​

API utilizada: Rick and Morty API — https://rickandmortyapi.com
Autenticación: Ninguna
Objetivo: Dominar map(), filter(), reduce() y transformación de datos JSON con una API real.

¿Qué vamos a hacer? ​

Una aplicación que carga personajes de Rick and Morty y permite filtrarlos y analizarlos usando los métodos de arrays de JavaScript.

Demostraremos:

  • filter() — Filtrar personajes por estado (vivo/muerto/desconocido)
  • map() — Extraer solo ciertos campos de cada personaje
  • reduce() — Calcular estadísticas (contar por especie, por género, etc.)
  • Transformación de estructuras — Adaptar el JSON de la API a nuestras necesidades

        // Variable global con todos los personajes
        let todosLosPersonajes = [];


        // -------------------------------------------------------
        // CARGAR DATOS DE LA API

        async function cargarPersonajes() {
            try {
                // La API tiene paginación, vamos a cargar las primeras 2 páginas
                const [pagina1, pagina2] = await Promise.all([
                    fetch('https://rickandmortyapi.com/api/character?page=1').then(r => r.json()),
                    fetch('https://rickandmortyapi.com/api/character?page=2').then(r => r.json())
                ]);

                // Combinar resultados
                const personajesCrudos = [...pagina1.results, ...pagina2.results];

                // TRANSFORMACIÓN: simplificar la estructura
                todosLosPersonajes = personajesCrudos.map(transformarPersonaje);

                // Mostrar todos inicialmente
                renderizar(todosLosPersonajes);
                calcularEstadisticas(todosLosPersonajes);
                document.getElementById('loading').style.display = 'none';

            } catch (error) {
                console.error('Error al cargar personajes:', error);
                document.getElementById('loading').innerHTML = 
                    '<p style="color: #d63d2e;">❌ Error al cargar datos de la API</p>';
            }
        }


        // -------------------------------------------------------
        // TRANSFORMACIÓN DE DATOS
        // Convertimos el JSON complejo de la API a una estructura simple
        // -------------------------------------------------------

        function transformarPersonaje(personaje) {
            return {
                id: personaje.id,
                nombre: personaje.name,
                estado: personaje.status.toLowerCase(),  // "Alive" → "alive"
                especie: personaje.species,
                genero: personaje.gender,
                origen: personaje.origin.name,
                imagen: personaje.image,
                // Campos calculados
                esHumano: personaje.species === 'Human',
                estaVivo: personaje.status === 'Alive'
            };
        }

        // -------------------------------------------------------
        // ESTADÍSTICAS CON reduce()
        // -------------------------------------------------------

        function calcularEstadisticas(personajes) {
            // 1. Contar por estado usando reduce()
            const porEstado = personajes.reduce((acumulador, personaje) => {
                acumulador[personaje.estado] = (acumulador[personaje.estado] || 0) + 1;
                return acumulador;
            }, {});

            // 2. Contar especies únicas
            const especies = personajes.reduce((acum, p) => {
                if (!acum.includes(p.especie)) acum.push(p.especie);
                return acum;
            }, []);

            // 3. Porcentaje de humanos
            const humanos = personajes.filter(p => p.esHumano).length;
            const porcentajeHumanos = ((humanos / personajes.length) * 100).toFixed(1);

            // Renderizar stats
            const statsHTML = `
                <div class="stat-card">
                    <div class="stat-value">${personajes.length}</div>
                    <div class="stat-label">Total</div>
                </div>
                <div class="stat-card">
                    <div class="stat-value">${porEstado.alive || 0}</div>
                    <div class="stat-label">Vivos</div>
                </div>
                <div class="stat-card">
                    <div class="stat-value">${porEstado.dead || 0}</div>
                    <div class="stat-label">Muertos</div>
                </div>
                <div class="stat-card">
                    <div class="stat-value">${especies.length}</div>
                    <div class="stat-label">Especies</div>
                </div>
                <div class="stat-card">
                    <div class="stat-value">${porcentajeHumanos}%</div>
                    <div class="stat-label">Humanos</div>
                </div>
            `;

            document.getElementById('stats').innerHTML = statsHTML;
        }


        // -------------------------------------------------------
        // RENDERIZADO CON map()
        // -------------------------------------------------------

        function renderizar(personajes) {
            const container = document.getElementById('personajes');

            // Usamos map() para convertir cada personaje en HTML
            const html = personajes.map(personaje => `
                <div class="personaje-card">
                    <img src="${personaje.imagen}" alt="${personaje.nombre}">
                    <div class="personaje-info">
                        <div class="personaje-nombre">${personaje.nombre}</div>
                        <div class="personaje-detalles">
                            <span>
                                <span class="estado ${personaje.estado}"></span>
                                ${traducirEstado(personaje.estado)}
                            </span>
                            <span>${personaje.especie}</span>
                        </div>
                    </div>
                </div>
            `).join('');

            container.innerHTML = html;
        }


        // -------------------------------------------------------
        // UTILIDADES
        // -------------------------------------------------------

        function traducirEstado(estado) {
            const traducciones = {
                'alive': 'Vivo',
                'dead': 'Muerto',
                'unknown': 'Desconocido'
            };
            return traducciones[estado] || estado;
        }

Conceptos clave demostrados ​

1. Transformación de estructura con map() ​

El JSON que devuelve la API tiene esta forma:

json
{
  "id": 1,
  "name": "Rick Sanchez",
  "status": "Alive",
  "species": "Human",
  "origin": {
    "name": "Earth (C-137)",
    "url": "https://..."
  }
}

Lo simplificamos a esto:

javascript
{
  id: 1,
  nombre: "Rick Sanchez",
  estado: "alive",
  especie: "Human",
  origen: "Earth (C-137)",
  esHumano: true,
  estaVivo: true
}

El código:

javascript
const personajesSimplificados = personajesCrudos.map(p => ({
    id: p.id,
    nombre: p.name,
    estado: p.status.toLowerCase(),
    especie: p.species,
    origen: p.origin.name,
    esHumano: p.species === 'Human'
}));

2. Filtrado con filter() ​

javascript
// Solo personajes vivos
const vivos = personajes.filter(p => p.estado === 'alive');

// Solo humanos
const humanos = personajes.filter(p => p.especie === 'Human');

// Combinación de filtros
const humanosVivos = personajes
    .filter(p => p.especie === 'Human')
    .filter(p => p.estado === 'alive');

3. Estadísticas con reduce() ​

Contar por categoría:

javascript
const porEstado = personajes.reduce((acum, p) => {
    acum[p.estado] = (acum[p.estado] || 0) + 1;
    return acum;
}, {});

// Resultado: { alive: 15, dead: 8, unknown: 2 }

Coleccionar valores únicos:

javascript
const especies = personajes.reduce((acum, p) => {
    if (!acum.includes(p.especie)) acum.push(p.especie);
    return acum;
}, []);

// Resultado: ["Human", "Alien", "Robot", ...]

4. Encadenamiento de métodos ​

javascript
const nombresDeMuertosHumanos = personajes
    .filter(p => p.estado === 'dead')      // 1. Solo muertos
    .filter(p => p.esHumano)               // 2. Solo humanos
    .map(p => p.nombre)                     // 3. Solo nombres
    .sort();                                // 4. Ordenar alfabéticamente

// Resultado: ["Abradolf Lincler", "Beth Smith", ...]

Ejercicios para clase ​

  1. Añade un nuevo filtro que muestre solo personajes cuyo origen sea "Earth (C-137)". Usa filter().

  2. Calcula el promedio de personajes por especie. Usa reduce() para contar cuántos hay de cada especie, luego divide el total entre el número de especies.

  3. Crea una lista de nombres únicos que empiecen por "R". Combina filter() y map().

  4. Encuentra la especie más común. Usa reduce() para contar por especie, luego encuentra la que tiene mayor valor.

Comparativa: forma imperativa vs funcional ​

❌ Forma imperativa (con bucles):

javascript
const vivos = [];
for (let i = 0; i < personajes.length; i++) {
    if (personajes[i].estado === 'alive') {
        vivos.push(personajes[i]);
    }
}

✅ Forma funcional (con filter):

javascript
const vivos = personajes.filter(p => p.estado === 'alive');

La segunda es más legible, menos propensa a errores y más fácil de encadenar con otras operaciones.
